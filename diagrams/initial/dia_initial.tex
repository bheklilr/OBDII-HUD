                \draw[myarrow] (comm.north) -- ++(0, 0.3) -| ([xshift=-1cm] obdII.south);
            \node (obdIIdata) [abstract, text width=4cm, rectangle split, rectangle split parts=2, right=of aux01]

                \node (obdIIbt) [abstract, text width=4cm, rectangle split, rectangle split parts=2, below=of comm]

        \node (display) [abstract, rectangle split, rectangle split parts=2, left=of obdII]
            {
                \textbf{Display}

            \draw[myarrow] (display.north) -- ++(0, 1) -| ([xshift=-1cm] obdIIhud.south);
